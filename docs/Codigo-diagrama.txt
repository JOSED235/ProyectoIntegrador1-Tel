    artifact "Firmware embebido\n- Lectura de sensores (100 Hz)\n- Buffer y empaquetado JSON\n- Control START/STOP (botón físico\n  o comando MQTT remoto)\n- Publica icesi/jose/esp32/data y /status\n- Suscrito a icesi/jose/esp32/control\n- HTTP POST /captura (vía redundante,\n  desactivada en el firmware activo)" as ESP_Art
}

node "Navegador Web" <<device>> {
    artifact "Cliente (SPA React)\n- Login y roles (admin/doctor/paciente)\n- Accionar captura (START/STOP)\n- Visualización en vivo y análisis\n- Cliente MQTT por WebSocket" as Browser_Art
}

' =========================
'  CAPA 3 — Broker MQTT público (Internet)
' =========================
cloud "Broker MQTT Público (broker.emqx.io)" <<broker>> {
    artifact "Tópicos\n- icesi/jose/esp32/data    -> lotes de muestras\n- icesi/jose/esp32/status  -> estado RECORDING/IDLE\n- icesi/jose/esp32/control -> comandos START/STOP" as Broker_Art
}

' =========================
'  CAPA 4 — Entorno Docker (servidor)
' =========================
node "Entorno Docker" <<execution environment>> {

    node "Contenedor Frontend" <<container>> {
        artifact "React 19 + Vite, servido por Nginx\n(puerto host 5173 -> 80)\n- Visualización de señales\n- Historial y gestión de usuarios\n- Cliente MQTT (broker.emqx.io)\n- Nginx: estáticos + proxy /api/" as Front_Art
    }

    node "Contenedor Backend" <<container>> {
        node "API FastAPI" {
            artifact "Endpoints (puerto 8001)\n- /auth/* (login, registro, JWT)\n- /users/doctors\n- /patients\n- POST /captura (HTTP redundante)\n- GET /sesiones, /captura/{id}" as API_Art
        }
        node "MQTT Subscriber (Paho)" {
            artifact "Hilo de fondo\n- Suscrito a icesi/jose/esp32/data\n- Vía primaria de persistencia" as Sub_Art
        }
        node "Módulo de validación y persistencia" {
            artifact "- Validación del payload\n- Resolución de paciente\n- Inserción de sesión / muestras\n- run_migrations() (ALTER TABLE)" as Pers_Art
        }
    }

    database "Contenedor PostgreSQL" <<container>> {
        artifact "postgres:17 (127.0.0.1:5432)\nTablas\n- users\n- patients\n- capture_sessions\n- sensor_samples" as DB_Art
    }
}

' =========================
'  NOTAS
' =========================
note right of "ESP32 DevKit"
  Flujo 1 (captura):
  Lee los 3 sensores, arma lotes
  JSON y los publica por MQTT al
  broker público (vía primaria).
end note

note right of "Contenedor Backend"
  Flujo 2 (persistencia):
  El MQTT Subscriber recibe los
  lotes desde el broker, valida
  el payload, resuelve el paciente
  y guarda los datos en PostgreSQL.
end note

note right of "Contenedor Frontend"
  Flujo 3 (visualización y control):
  El navegador usa MQTT por
  WebSocket para ver el estado en
  vivo y enviar START/STOP, y
  consulta la API REST (vía Nginx
  /api/) para historial y análisis.
end note
